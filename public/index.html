1. Merge Intervals

Problem: Merge overlapping intervals.

Input: [[1,3],[2,6],[8,10],[15,18]]

Output: [[1,6],[8,10],[15,18]]

fun merge(intervals: Array): List {
    intervals.sortBy { it[0] }
    val result = mutableListOf()
    var current = intervals[0]

    for (i in 1 until intervals.size) {
        val next = intervals[i]
        if (next[0] <= current[1]) {
            current[1] = maxOf(current[1], next[1])
        } else {
            result.add(current)
            current = next
        }
    }
    result.add(current)
    return result
}

2. Meeting Rooms II

Problem: Find minimum meeting rooms required.

Input: [[0,30],[5,10],[15,20]]

Output: 2

fun minMeetingRooms(intervals: Array): Int {
    intervals.sortBy { it[0] }
    val heap = java.util.PriorityQueue()

    for (interval in intervals) {
        if (heap.isNotEmpty() && heap.peek() <= interval[0]) heap.poll()
        heap.offer(interval[1])
    }
    return heap.size
}

3. Non-overlapping Intervals

Problem: Find minimum intervals to remove to eliminate overlaps.

Input: [[1,2],[2,3],[3,4],[1,3]]

Output: 1

fun eraseOverlapIntervals(intervals: Array): Int {
    intervals.sortBy { it[1] }
    var count = 0
    var prevEnd = intervals[0][1]
    for (i in 1 until intervals.size) {
        if (intervals[i][0] < prevEnd) count++
        else prevEnd = intervals[i][1]
    }
    return count
}

5. Interval Intersection

Problem: Find intersections between two interval lists.

Input: [[0,2],[5,10]] & [[1,5],[8,12]]

Output: [[1,2],[5,5],[8,10]]

fun intervalIntersection(A: Array, B: Array): List {
    val result = mutableListOf()
    var i = 0
    var j = 0
    while (i < A.size && j < B.size) {
        val start = maxOf(A[i][0], B[j][0])
        val end = minOf(A[i][1], B[j][1])
        if (start <= end) result.add(intArrayOf(start, end))
        if (A[i][1] < B[j][1]) i++ else j++
    }
    return result
}

6. Minimum Arrows

Problem: Minimum arrows to burst balloons.

Input: [[10,16],[2,8],[1,6],[7,12]]

Output: 2

fun findMinArrowShots(points: Array): Int {
    points.sortBy { it[1] }
    var arrows = 1
    var end = points[0][1]
    for (i in 1 until points.size) {
        if (points[i][0] > end) {
            arrows++
            end = points[i][1]
        }
    }
    return arrows
}

7. Employee Free Time

Problem: Given schedules of employees, return common free time intervals.

Input: [[[1,2],[5,6]], [[1,3]], [[4,10]]]

Output: [[3,4]]

fun employeeFreeTime(schedule: List>): List {
    val all = schedule.flatten().toTypedArray()
    all.sortBy { it[0] }
    val merged = mutableListOf()
    var cur = all[0]
    for (i in 1 until all.size) {
        val n = all[i]
        if (n[0] <= cur[1]) cur[1] = maxOf(cur[1], n[1])
        else { merged.add(cur); cur = n }
    }
    merged.add(cur)
    val res = mutableListOf()
    for (i in 1 until merged.size) {
        res.add(intArrayOf(merged[i-1][1], merged[i][0]))
    }
    return res
}

8. Remove Covered Intervals

Problem: Remove intervals covered by another.

Input: [[1,4],[3,6],[2,8]]

Output: 2

fun removeCoveredIntervals(intervals: Array): Int {
    intervals.sortWith(compareBy({ it[0] }, { -it[1] }))
    var count = 0
    var end = 0
    for (it in intervals) {
        if (it[1] > end) { count++; end = it[1] }
    }
    return count
}

9. Summary Ranges

Problem: Convert sorted array into range strings.

Input: [0,1,2,4,5,7]

Output: ["0->2","4->5","7"]

fun summaryRanges(nums: IntArray): List {
    val res = mutableListOf()
    var start = nums[0]
    for (i in 1..nums.size) {
        if (i == nums.size || nums[i] != nums[i-1] + 1) {
            if (start == nums[i-1]) res.add("$start")
            else res.add("$start->${nums[i-1]}")
            if (i < nums.size) start = nums[i]
        }
    }
    return res
}

10. Partition Labels

Problem: Partition string so each char appears in one part.

Input: ababcbacadefegdehijhklij

Output: [9,7,8]

fun partitionLabels(s: String): List {
    val last = IntArray(26)
    for (i in s.indices) last[s[i]-'a'] = i
    val res = mutableListOf()
    var start = 0; var end = 0
    for (i in s.indices) {
        end = maxOf(end, last[s[i]-'a'])
        if (i == end) { res.add(end-start+1); start = i+1 }
    }
    return res
}

11. Task Scheduler

Problem: Schedule tasks with cooldown n.

Input: [A,A,A,B,B,B], n=2

Output: 8

fun leastInterval(tasks: CharArray, n: Int): Int {
    val freq = IntArray(26)
    for (t in tasks) freq[t-'A']++
    val pq = java.util.PriorityQueue(compareByDescending{it})
    freq.filter{it>0}.forEach{pq.add(it)}
    var time=0
    while (pq.isNotEmpty()) {
        val temp = mutableListOf()
        for (i in 0..n) {
            if (pq.isNotEmpty()) {
                val c = pq.poll()-1
                if (c>0) temp.add(c)
            }
            time++
            if (pq.isEmpty() && temp.isEmpty()) break
        }
        temp.forEach{pq.add(it)}
    }
    return time
}

13. Car Pooling

Problem: Check if capacity exceeded.

Input: [[2,1,5],[3,3,7]], cap=4

Output: false

fun carPooling(trips: Array, cap: Int): Boolean {
    val map = java.util.TreeMap()
    for (t in trips) {
        map[t[1]] = map.getOrDefault(t[1],0)+t[0]
        map[t[2]] = map.getOrDefault(t[2],0)-t[0]
    }
    var curr=0
    for (v in map.values) {
        curr+=v
        if (curr>cap) return false
    }
    return true
}

14. Maximum Overlap

Problem: Find max overlapping intervals.

Input: [[1,3],[2,5],[4,6]]

Output: 2

fun maxOverlap(intervals: Array): Int {
    val map = java.util.TreeMap()
    for (it in intervals) {
        map[it[0]] = map.getOrDefault(it[0],0)+1
        map[it[1]] = map.getOrDefault(it[1],0)-1
    }
    var max=0; var curr=0
    for (v in map.values) {
        curr+=v
        max = maxOf(max,curr)
    }
    return max
}
